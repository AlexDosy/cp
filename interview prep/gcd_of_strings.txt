class Solution(object):
    def gcdOfStrings(self, str1, str2):
        """
        :type str1: str
        :type str2: str
        :rtype: str
        """
        if not (str1.startswith(str2) or str2.startswith(str1)):
            return ""
        a,b = str1,str2
        while a and b:
            if a.startswith(b):
                a=a[len(b):]
            elif b.startswith(a):
                b= b[len(a):]
            else:
                return ""
        return a or b
