class Solution(object):
    def kidsWithCandies(self, candies, extraCandies):
        """
        :type candies: List[int]
        :type extraCandies: int
        :rtype: List[bool]
        """
        res=[]
        max_candy=max(candies)
        for i in candies:
            

            res.append(i+extraCandies>=max_candy)

        return res
