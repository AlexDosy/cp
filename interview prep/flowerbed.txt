class Solution(object):
    def canPlaceFlowers(self, flowerbed, n):
        i=0
        l= len(flowerbed)
        
        while i<l and n>0:
            if flowerbed[i]==0:
                emptyleft = (i==0 or flowerbed[i-1]==0)
                emptyright = (i==l-1 or flowerbed[i+1]==0)
                if emptyleft and emptyright:
                    flowerbed[i]=1
                    n-=1
                    i+=2
                    continue
            i+=1
        return n==0
