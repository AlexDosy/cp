def setZeroes(matrix):
    m, n = len(matrix), len(matrix[0])
    rows = [False] * m
    cols = [False] * n

    # First pass: record rows and columns with zero
    for i in range(m):
        for j in range(n):
            if matrix[i][j] == 0:
                rows[i] = True
                cols[j] = True

    # Second pass: set zeros
    for i in range(m):
        for j in range(n):
            if rows[i] or cols[j]:
                matrix[i][j] = 0
