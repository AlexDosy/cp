class Solution(object):
    def combinationSum3(self, k, n):
        """
        :type k: int
        :type n: int
        :rtype: List[List[int]]
        """
        res =[]

        def backtrack(start, curr, total):
            if len(curr) ==k:
                if total ==n:
                    res.append(curr[:])
                return 
            for num in range(start, 10):
                if total+num >n:
                    break
                curr.append(num)
                backtrack(num+1, curr, total+num)
                curr.pop()
        backtrack(1,[],0)
        return res
