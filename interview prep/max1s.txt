function longestOnes(nums: number[], k: number): number {
    let left =0;
    let maxLength=0;
    let zeroesCount =0
    for (let right=0;right<nums.length; right++){
        if(nums[right]===0){
            zeroesCount++
        }
        while (zeroesCount>k){
            if(nums[left]===0){
                zeroesCount--;
            }
            left++;
        }
        maxLength=Math.max(maxLength,right-left+1);
    }
    return maxLength;
};

right expands the window (adds new elements).

If the number of zeros in the window exceeds k, we shrink the window from the left until it's valid again.
