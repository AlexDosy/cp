class Solution(object):
    def minCostClimbingStairs(self, cost):
        n = len(cost)

        # dp[i] = min cost to reach step i
        # We only need last two states
        a, b = 0, 0   # dp[0], dp[1]

        for i in range(2, n + 1):
            curr = min(a + cost[i - 2], b + cost[i - 1])
            a, b = b, curr

        return b
