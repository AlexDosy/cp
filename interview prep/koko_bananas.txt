import math

class Solution(object):
    def minEatingSpeed(self, piles, h):
        l, r = 1, max(piles)

        while l <= r:
            k = (l + r) // 2
            hours = 0

            for p in piles:
                hours += (p + k - 1) // k   # faster integer ceil

            if hours <= h:
                r = k - 1
            else:
                l = k + 1

        return l   # l is the minimum feasible speed
