class MinStack:

    def __init__(self):
        self.stack = []
        self.min_stack = []

    def push(self, val: int) -> None:
        self.stack.append(val)
        if not self.min_stack:
            self.min_stack.append(val)
        else:
            self.min_stack.append(min(val, self.min_stack[-1]))

    def pop(self) -> None:
        self.stack.pop()
        self.min_stack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.min_stack[-1]



Invariant:
min_stack[i] = minimum of stack[0..i]

Operations

push(val)
Push val to stack.
Push min(val, current_min) to min_stack.

pop()
Pop from both stacks.

top()
Return top of stack.

getMin()
Return top of min_stack
