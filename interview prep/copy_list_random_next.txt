# Definition for a Node.
# class Node:
#     def __init__(self, x, next=None, random=None):
#         self.val = int(x)
#         self.next = next
#         self.random = random

class Solution(object):
    def copyRandomList(self, head):
        if not head:
            return None

        # 1️⃣ Insert copied nodes after originals
        curr = head
        while curr:
            copy = Node(curr.val)
            copy.next = curr.next
            curr.next = copy
            curr = copy.next

        # 2️⃣ Assign random pointers
        curr = head
        while curr:
            if curr.random:
                curr.next.random = curr.random.next
            curr = curr.next.next

        # 3️⃣ Separate the lists
        dummy = Node(0)
        copy_curr = dummy
        curr = head

        while curr:
            copy = curr.next
            curr.next = copy.next
            copy_curr.next = copy

            copy_curr = copy
            curr = curr.next

        return dummy.next

Every copied node is placed next to its original

Random pointers can be assigned without a hash map

Final separation cleanly restores the original list
