class StockSpanner:
    def __init__(self):
        self.stack = []   # each element: (price, span)

    def next(self, price):
        span = 1
        
        # pop all previous prices less than or equal to today's price
        while self.stack and self.stack[-1][0] <= price:
            span += self.stack[-1][1]
            self.stack.pop()
        
        self.stack.append((price, span))
        return span
