// Definition for a binary tree node.
class TreeNode {
  val: number;
  left: TreeNode | null;
  right: TreeNode | null;
  constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
    this.val = val === undefined ? 0 : val;
    this.left = left === undefined ? null : left;
    this.right = right === undefined ? null : right;
  }
}

function longestZigZag(root: TreeNode | null): number {
  let maxLen = 0;

  function dfs(node: TreeNode | null, isLeft: boolean, length: number): void {
    if (!node) return;

    // Update maximum length found so far
    maxLen = Math.max(maxLen, length);

    // If last move was to the left, now move right, and vice versa
    if (node.left) dfs(node.left, false, isLeft ? length + 1 : 1);
    if (node.right) dfs(node.right, true, isLeft ? 1 : length + 1);
  }

  // Start DFS from root, exploring both initial directions
  dfs(root, true, 0);
  dfs(root, false, 0);

  return maxLen;
}
